Here is what I have decided on, in case this is at all useful

function forward_diff_i(f)
    ∇ᵢ₊ = padarray(f, Pad(1, 1))

    for j in 1:size(f, 2), i in 1:size(f, 1)
        ∇ᵢ₊[i, j] = ∇ᵢ₊[i + 1, j] - ∇ᵢ₊[i, j]
    end

    return parent(∇ᵢ₊[1:end-1, 1:end-1])
end
