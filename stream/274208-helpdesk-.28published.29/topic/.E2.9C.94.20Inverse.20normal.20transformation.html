using Distributions
using CairoMakie
using StatsFuns
using StatsBase

function invnormaltransform(v; c=3/8)
    (μ, σ) = mean_and_var(v)
    rank = invperm(sortperm(v))
    N = length(v)
    return [norminvcdf(μ, σ, (x - c) / (N - 2c + 1)) for x in rank]
end

x = rand(Beta(3, 1), 1000); y = invnormaltransform(x);

hist(x)
hist!(y)
current_figure()
