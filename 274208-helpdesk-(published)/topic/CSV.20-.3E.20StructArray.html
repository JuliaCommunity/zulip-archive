Mason Protter (Feb 03 2021 at 03:26):

You could do something like this:

julia> function csv_to_SA(T, file::String)
           csv = CSV.File(file)
           SA = StructVector([T(csv[1])])
           for row in csv[2:end]
               push!(SA, T(row))
           end
           SA
       end

julia> csv_to_SA(NamedTuple, "foo.csv")
